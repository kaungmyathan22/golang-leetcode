func maxRotateFunction(nums []int) int {
    result := 0
    rep := len(nums)
    for i:=0; i < rep; i++ {
        fmt.Println(nums)
        nums = append(nums[rep-1],nums[1:])
        curResult := 0
        for idx, v := range nums  {
            curResult += idx * v
        }
        if curResult > result {
            result = curResult
        }
    }
    return result
}

package main

// [1,4], 2 // 1,2 4,5
// [1,2], 2 // 1,2, 2,3

func findPoisonedDuration(timeSeries []int, duration int) int {

	result := []int{}

	for _,v := range timeSeries {



		if result[len(result)-1] == v {

			result = append(result, v + duration - 1)

		} else {

		    result = append(result, []int{v,v + duration - 1}...)

		}

	}

	return len(result)

}

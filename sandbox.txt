/**
try every possible substring length from 1 to len(s)/2:
    if length divides len(s) evenly:
        repeat that substring len(s)/length times
        does it equal s?
        if yes → return true

return false
*/

func repeatedSubstringPattern(s string) bool {
    substrLen := len(s)/2
    for i:=1 ; i < substrLen ; i++ {
        if len(s) % i == 0  {
            comp := strings.Repeat(s[:i], len(s)/i)
            if comp == s {
                return true
            }
        }
    }
    return false
}
